from flask import Flask, request, render_template_string from textblob import TextBlob app = Flask(__name__) html = """ Sentiment Analyzer AI

AI Sentiment Analyzer


{% if result %}
Sentiment: {{ result }}
{% endif %} """ @app.route("/", methods=["GET", "POST"]) def home(): result = "" css_class = "" if request.method == "POST": text = request.form["text"] analysis = TextBlob(text) polarity = analysis.sentiment.polarity if polarity > 0: result = "Positive 😊" css_class = "positive" elif polarity < 0: result = "Negative 😞" css_class = "negative" else: result = "Neutral 😐" css_class = "neutral" return render_template_string(html, result=result, css_class=css_class) if __name__ == "__main__": app.run(debug=True)
